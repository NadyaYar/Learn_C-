bool isLeap(int year) {
if (year % 4 != 0) { // Рік, що не кратний 4, не є високосним
return false;
} else if (year % 100 != 0) { // Рік, що кратний 4 і не кратний 100, є високосним
return true;
} else if (year % 400 != 0) { // Рік, що кратний 100 і не кратний 400, не є високосним
return false;
} else { // Рік, що кратний 400, є високосним
return true;
}
}
