// перший способ за допомогою switch
int monthLength(int year, int month) {
    int days;
    switch(month) {
        case 2:
            if(isLeap(year))
                days = 29;
            else
                days = 28;
            break;
        case 4:
        case 6:
        case 9:
        case 11:
            days = 30;
            break;
        default:
            days = 31;
    }
    return days;
}
