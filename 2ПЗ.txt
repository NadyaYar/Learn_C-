#include <iostream>
#include <cstdlib> // для функції rand()

using namespace std;

const int ROWS = 5;
const int COLS = 5;

int main() {
    int matrix[ROWS][COLS];
    int *ptr;
    int even_count = 0, odd_count = 0;

    // Заповнення масиву випадковими значеннями
    ptr = &matrix[0][0];
    for (int i = 0; i < ROWS * COLS; i++) {
        *(ptr + i) = rand() % 100 - 50;
        if (*(ptr + i) % 2 == 0) {
            even_count++;
        } else {
            odd_count++;
        }
    }

    // Виведення елементів на головній діагоналі
    cout << "Elements on the main diagonal:\n";
    ptr = &matrix[0][0];
    for (int i = 0; i < ROWS; i++) {
        cout << *(ptr + i * COLS + i) << " ";
    }
    cout << endl;

    // Виведення елементів на побічній діагоналі
    cout << "Elements on the secondary diagonal:\n";
    ptr = &matrix[0][0];
    for (int i = 0; i < ROWS; i++) {
        cout << *(ptr + i * COLS + (COLS - 1 - i)) << " ";
    }
    cout << endl;

    // Виведення кількості парних та непарних елементів масиву
    cout << "Number of even elements: " << even_count << endl;
    cout << "Number of odd elements: " << odd_count << endl;

    return 0;
}
