#include <iostream>
#include <cstdlib>
#include <ctime> 

using namespace std;

int main() {
    // Запустимо генератор випадкових чисел
    srand(time(0));
    
    // Введіть розмір масиву
    int n;
    cout << "Введіть розмір масиву: ";
    cin >> n;
    
    // Створюємо динамічний масив
    double* arr = new double[n];
    
    // Заповнюємо масив випадковими значеннями
    for (int i = 0; i < n; i++) {
        arr[i] = ((double)rand() / RAND_MAX) * 4 - 2; // генеруємо число в діапазоні (-2, 2)
    }
    
    // Рахуємо кількість додатніх та від'ємних чисел
    int positive = 0, negative = 0;
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0) {
            positive++;
        } else if (arr[i] < 0) {
            negative++;
        }
    }
    
    // Виводимо результат
    cout << "Кількість додатніх чисел: " << positive << endl;
    cout << "Кількість від'ємних чисел: " << negative << endl;
    
    // Звільняємо пам'ять
    delete[] arr;
    
    return 0;
}
